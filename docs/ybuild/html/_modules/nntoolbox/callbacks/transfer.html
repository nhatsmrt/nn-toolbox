def freeze_bn(module: Module):
    for submodule in module.modules():
        for bn_type in BN_TYPE:
            if isinstance(submodule, bn_type):
                if not next(submodule.parameters()).requires_grad:
                    submodule.eval()
        freeze_bn(submodule)
